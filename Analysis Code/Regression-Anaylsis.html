# Regression model 1
Model1 <- lm(formula = index_tot ~ College_Scorecard + high_earning + time_v + College_Scorecard*time_v + College_Scorecard*high_earning, data = My_Cleaned_Data)
export_summs(Model1, digits = 3, robust = TRUE)
